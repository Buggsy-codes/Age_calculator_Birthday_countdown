from datetime import date
today = date.today()
birth_year, birth_month, birth_day = map(int, input(Enter YYYY MM DD ).split())
age = today.year - birth_year - ((today.month, today.day) <(birth_month, birth_day))
print(fYour age : {age})
target_year = today.year + ((birth_month, birth_day) < (today.month, today.day))
try:
    next_bday = date(target_year, birth_month, birth_day)
except ValueError:
    next_bday = date(target_year, 3, 1)
days_left = next_bday - today
print(fDays left till next birthday: {days_left.days})
